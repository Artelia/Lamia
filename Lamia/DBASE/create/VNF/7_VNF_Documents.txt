#EXPO SELECT Objet.datecreation AS date_document, Objet.commentaire AS description, Photo.id_list_photo, Photo.nom_fichier AS nom_document, Photo.content_type AS type_doc, Photo.id_photo AS id_fichier, Photo.id_list_document, Ressource.id_ressource AS id_document FROM Ressource INNER JOIN Photo ON Photo.id_ressource = Ressource.id_ressource INNER JOIN Objet ON Objet.id_objet = Ressource.id_objet
