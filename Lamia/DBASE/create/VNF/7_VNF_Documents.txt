#EXPO SELECT Objet.datecreation AS date_document, Objet.commentaire AS description, Photo.list_photo AS id_lst_photo, REPLACE (Ressource.file, RTRIM(Ressource.file, REPLACE(Ressource.file, '\', '')),'') AS nom_document, Photo.id_photo AS id_fichier, Photo.list_document AS id_lst_document, Ressource.contactnom AS proprietaire, Ressource.id_ressource AS id_document FROM Ressource INNER JOIN Photo ON Photo.id_ressource = Ressource.id_ressource INNER JOIN Objet ON Objet.id_objet = Ressource.id_objet
