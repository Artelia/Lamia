#EXPO SELECT Propriete.id_propriete_option AS id_ouvrage_propriete, Propriete.propriete AS id_propriete, Equipement.partie AS id_partie, Equipement.equipement AS id_equipement, Infralineaire.id_ouvrage as num_ouvrage, Equipement.pk_debut, Equipement.pk_fin, Propriete.valeur, Equipement.groupe_number AS Groupe_propriete FROM Propriete_option INNER JOIN Equipement ON Equipement.id_equipement = lk_equipement INNER JOIN Infralineaire ON Equipement.lk_infralineaire = id_infralineaire
